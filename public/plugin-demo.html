Integration Code Example

// Initialize MedVerse SDK
const medverse = new MedVerseSDK({
  apiKey: 'your-api-key-here',
  environment: 'production',
  fhirEndpoint: 'https://your-ehr.com/fhir/r4',
  theme: 'auto'
});

// Create prescription widget
const prescriptionWidget = medverse.createPrescriptionWidget({
  container: '#prescription-container',
  patientId: 'patient-123',
  onPrescriptionCreated: (prescription) => {
    console.log('Prescription created:', prescription);
    // Sync with your EHR system
    updateEHRPrescription(prescription);
  }
});

// Create medical image interpreter
const imageWidget = medverse.createImageInterpreterWidget({
  container: '#image-container',
  supportedFormats: ['jpg', 'png', 'dicom'],
  onInterpretationComplete: (result) => {
    console.log('Image interpretation:', result);
    // Add to patient record
    addToPatientRecord(result);
  }
});

// Create voice assistant
const voiceWidget = medverse.createVoiceAssistantWidget({
  container: '#voice-container',
  autoStart: false,
  onCommandExecuted: (command, result) => {
    console.log('Voice command executed:', command, result);
  }
});

// FHIR Integration
medverse.syncWithFHIR({
  resourceType: 'Patient',
  id: 'patient-123',
  // ... FHIR patient data
});
